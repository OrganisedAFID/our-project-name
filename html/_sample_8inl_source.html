57 void Sample::Setup()
58 {
59  // Modify engine startup parameters
60  engineParameters_[EP_WINDOW_TITLE] = GetTypeName();
61  engineParameters_[EP_LOG_NAME] = GetSubsystem<FileSystem>()->GetAppPreferencesDir("urho3d", "logs") + GetTypeName() + ".log";
62  engineParameters_[EP_FULL_SCREEN] = false;
63  engineParameters_[EP_HEADLESS] = false;
64  engineParameters_[EP_SOUND] = false;
65 
66  // Construct a search path to find the resource prefix with two entries:
67  // The first entry is an empty path which will be substituted with program/bin directory -- this entry is for binary when it is still in build tree
68  // The second and third entries are possible relative paths from the installed program/bin directory to the asset directory -- these entries are for binary when it is in the Urho3D SDK installation location
69  if (!engineParameters_.Contains(EP_RESOURCE_PREFIX_PATHS))
70  engineParameters_[EP_RESOURCE_PREFIX_PATHS] = ";../share/Resources;../share/Urho3D/Resources";
71 }

104 void Sample::InitTouchInput()
105 {
106  touchEnabled_ = true;
107 
108  ResourceCache* cache = GetSubsystem<ResourceCache>();
109  Input* input = GetSubsystem<Input>();
110  XMLFile* layout = cache->GetResource<XMLFile>("UI/ScreenJoystick_Samples.xml");
111  const String& patchString = GetScreenJoystickPatchString();
112  if (!patchString.Empty())
113  {
114  // Patch the screen joystick layout further on demand
115  SharedPtr<XMLFile> patchFile(new XMLFile(context_));
116  if (patchFile->FromString(patchString))
117  layout->Patch(patchFile);
118  }
119  screenJoystickIndex_ = (unsigned)input->AddScreenJoystick(layout, cache->GetResource<XMLFile>("UI/DefaultStyle.xml"));
120  input->SetScreenJoystickVisible(screenJoystickSettingsIndex_, true);
121 }

156 void Sample::CreateLogo()
157 {
158  // Get logo texture
159  ResourceCache* cache = GetSubsystem<ResourceCache>();
160  Texture2D* logoTexture = cache->GetResource<Texture2D>("Textures/FishBoneLogo.png");
161  if (!logoTexture)
162  return;
163 
164  // Create logo sprite and add to the UI layout
165  UI* ui = GetSubsystem<UI>();
166  logoSprite_ = ui->GetRoot()->CreateChild<Sprite>();
167 
168  // Set logo sprite texture
169  logoSprite_->SetTexture(logoTexture);
170 
171  int textureWidth = logoTexture->GetWidth();
172  int textureHeight = logoTexture->GetHeight();
173 
174  // Set logo sprite scale
175  logoSprite_->SetScale(256.0f / textureWidth);
176 
177  // Set logo sprite size
178  logoSprite_->SetSize(textureWidth, textureHeight);
179 
180  // Set logo sprite hot spot
181  logoSprite_->SetHotSpot(textureWidth, textureHeight);
182 
183  // Set logo sprite alignment
184  logoSprite_->SetAlignment(HA_RIGHT, VA_BOTTOM);
185 
186  // Make logo not fully opaque to show the scene underneath
187  logoSprite_->SetOpacity(0.9f);
188 
189  // Set a low priority for the logo so that other UI elements can be drawn on top
190  logoSprite_->SetPriority(-100);

245 void Sample::HandleKeyDown(StringHash /*eventType*/, VariantMap& eventData)
246 {
247  using namespace KeyDown;
248 
249  int key = eventData[P_KEY].GetInt();
250 
251  // Toggle console with F1
252  if (key == KEY_F1)
253  GetSubsystem<Console>()->Toggle();
254 
255  // Toggle debug HUD with F2
256  else if (key == KEY_F2)
257  GetSubsystem<DebugHud>()->ToggleAll();
258 
259  // Common rendering quality controls, only when UI has no focused element
260  else if (!GetSubsystem<UI>()->GetFocusElement())
261  {
262  Renderer* renderer = GetSubsystem<Renderer>();
263 
264  // Preferences / Pause
265  if (key == KEY_SELECT && touchEnabled_)
266  {
267  paused_ = !paused_;
268 
269  Input* input = GetSubsystem<Input>();
270  if (screenJoystickSettingsIndex_ == M_MAX_UNSIGNED)
271  {
272  // Lazy initialization
273  ResourceCache* cache = GetSubsystem<ResourceCache>();
274  screenJoystickSettingsIndex_ = (unsigned)input->AddScreenJoystick(cache->GetResource<XMLFile>("UI/ScreenJoystickSettings_Samples.xml"), cache->GetResource<XMLFile>("UI/DefaultStyle.xml"));
275  }
276  else
277  input->SetScreenJoystickVisible(screenJoystickSettingsIndex_, paused_);
278  }
279 
280  // Texture quality
281  else if (key == '1')
282  {
283  auto quality = (unsigned)renderer->GetTextureQuality();
284  ++quality;
285  if (quality > QUALITY_HIGH)
286  quality = QUALITY_LOW;
287  renderer->SetTextureQuality((MaterialQuality)quality);
288  }
289 
290  // Material quality
291  else if (key == '2')
292  {
293  auto quality = (unsigned)renderer->GetMaterialQuality();
294  ++quality;
295  if (quality > QUALITY_HIGH)
296  quality = QUALITY_LOW;
297  renderer->SetMaterialQuality((MaterialQuality)quality);
298  }
299 
300  // Specular lighting
301  else if (key == '3')
302  renderer->SetSpecularLighting(!renderer->GetSpecularLighting());
303 
304  // Shadow rendering
305  else if (key == '4')
306  renderer->SetDrawShadows(!renderer->GetDrawShadows());
307 
308  // Shadow map resolution
309  else if (key == '5')
310  {
311  int shadowMapSize = renderer->GetShadowMapSize();
312  shadowMapSize *= 2;
313  if (shadowMapSize > 2048)
314  shadowMapSize = 512;
315  renderer->SetShadowMapSize(shadowMapSize);
316  }
317 
318  // Shadow depth and filtering quality
319  else if (key == '6')
320  {
321  ShadowQuality quality = renderer->GetShadowQuality();
322  quality = (ShadowQuality)(quality + 1);
323  if (quality > SHADOWQUALITY_BLUR_VSM)
324  quality = SHADOWQUALITY_SIMPLE_16BIT;
325  renderer->SetShadowQuality(quality);
326  }
327 
328  // Occlusion culling
329  else if (key == '7')
330  {
331  bool occlusion = renderer->GetMaxOccluderTriangles() > 0;
332  occlusion = !occlusion;
333  renderer->SetMaxOccluderTriangles(occlusion ? 5000 : 0);
334  }
335 
336  // Instancing
337  else if (key == '8')
338  renderer->SetDynamicInstancing(!renderer->GetDynamicInstancing());
339 
340  // Take screenshot
341  else if (key == '9')
342  {
343  Graphics* graphics = GetSubsystem<Graphics>();
344  Image screenshot(context_);
345  graphics->TakeScreenShot(screenshot);
346  // Here we save in the Data folder with date and time appended
347  screenshot.SavePNG(GetSubsystem<FileSystem>()->GetProgramDir() + "Data/Screenshot_" +
348  Time::GetTimeStamp().Replaced(':', '_').Replaced('.', '_').Replaced(' ', '_') + ".png");
349  }
350  }
351 }

353 void Sample::HandleSceneUpdate(StringHash /*eventType*/, VariantMap& eventData)
354 {
355  // Move the camera by touch, if the camera node is initialized by descendant sample class
356  if (touchEnabled_ && cameraNode_)
357  {
358  Input* input = GetSubsystem<Input>();
359  for (unsigned i = 0; i < input->GetNumTouches(); ++i)
360  {
361  TouchState* state = input->GetTouch(i);
362  if (!state->touchedElement_) // Touch on empty space
363  {
364  if (state->delta_.x_ ||state->delta_.y_)
365  {
366  Camera* camera = cameraNode_->GetComponent<Camera>();
367  if (!camera)
368  return;
369 
370  Graphics* graphics = GetSubsystem<Graphics>();
371  yaw_ += TOUCH_SENSITIVITY * camera->GetFov() / graphics->GetHeight() * state->delta_.x_;
372  pitch_ += TOUCH_SENSITIVITY * camera->GetFov() / graphics->GetHeight() * state->delta_.y_;
373 
374  // Construct new orientation for the camera scene node from yaw and pitch; roll is fixed to zero
375  cameraNode_->SetRotation(Quaternion(pitch_, yaw_, 0.0f));
376  }
377  else
378  {
379  // Move the cursor to the touch position
380  Cursor* cursor = GetSubsystem<UI>()->GetCursor();
381  if (cursor && cursor->IsVisible())
382  cursor->SetPosition(state->position_);
383  }
384  }
385  }
386  }
387 }
